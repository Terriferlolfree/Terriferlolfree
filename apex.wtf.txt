function solve(angle, radius)
    return Vector2.new(
        math.sin(math.rad(angle)) * radius,
        math.cos(math.rad(angle)) * radius
    )
end

    if _tick - last_render > crosshair.refreshrate then
        last_render = _tick

        local position = (
            crosshair.mode == 'center' and camera.ViewportSize / 2 or

        )

local function setLockedTarget(plr)

local function isAlive(p)
    if not p or not p.Character then return false end
    local hum = p.Character:FindFirstChildOfClass("Humanoid")
    return hum and hum.Health > 0
end

-- get aim part based on dropdown
local function getAimPart(char)
    if not char then return nil end
    local selected = HitPartDropdown.Value
    if selected == "Head" and char:FindFirstChild("Head") then return char.Head end
    if selected == "HumanoidRootPart" and char:FindFirstChild("HumanoidRootPart") then return char.HumanoidRootPart end
    if selected == "Torso" and char:FindFirstChild("Torso") then return char.Torso end
    if selected == "UpperTorso" and char:FindFirstChild("UpperTorso") then return char.UpperTorso end
    return char.PrimaryPart or char:FindFirstChildWhichIsA("BasePart")
end

local function calculateTargetPos(aimPart)
    local pred = tonumber(PredictionBox.Value) or 0
    local smooth = tonumber(SmoothnessBox.Value) or 0.8
    local jump = tonumber(JumpOffsetBox.Value) or 0.09
    local vel = aimPart.Velocity or Vector3.new(0,0,0)
    local predictedPos = aimPart.Position + vel * pred + Vector3.new(0, jump, 0)
    return predictedPos, smooth
end

local function screenDistanceFromCenter(vec2)
    local vsize = Camera.ViewportSize
    local center = Vector2.new(vsize.X/2, vsize.Y/2)
    return (vec2 - center).Magnitude
end

local function findBestTarget()
    local best, bestDist = nil, math.huge
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and isAlive(p) then
            local char = p.Character
            local aim = getAimPart(char)
            if aim then
                local proj, onScreen = Camera:WorldToViewportPoint(aim.Position)
                if onScreen then
                    local dist = screenDistanceFromCenter(Vector2.new(proj.X, proj.Y))
                    local viewportSize = Camera.ViewportSize
                    local maxDist = math.max(viewportSize.X, viewportSize.Y) * (DEFAULTS.FOV / 180)
                    if dist <= maxDist then
                        if WallCheckToggle.Value then
                            local origin = Camera.CFrame.Position
                            local dir = aim.Position - origin
                            local rp = RaycastParams.new()
                            rp.FilterDescendantsInstances = {LocalPlayer.Character}
                            rp.FilterType = Enum.RaycastFilterType.Blacklist
                            rp.IgnoreWater = true
                            local rc = Workspace:Raycast(origin, dir, rp)
                            if rc and rc.Instance and rc.Instance:IsDescendantOf(char) then
                                if dist < bestDist then best = p; bestDist = dist end
                            end
                        else
                            if dist < bestDist then best = p; bestDist = dist end
                        end
                    end
                end
            end
        end
    end
    return best
end

-- ----------------- RenderStepped Loop -----------------
local lastTick = tick()
local conn = RunService.RenderStepped:Connect(function(dt)
    -- button size update
    local bs = ton(ButtonSizeSlider.Value, DEFAULTS.ButtonSize)
    if ButtonFrame.Size.X.Offset ~= bs then ButtonFrame.Size = UDim2.new(0, bs, 0, bs) end

    if not CamlockEnabled.Value then
        if locked then setLockedTarget(nil) end
        return
    end
    if not locked or not lockedTarget or not isAlive(lockedTarget) then
        if locked then setLockedTarget(nil) end
        return
    end

    local char = lockedTarget.Character
    if not char then setLockedTarget(nil); return end
    local aim = getAimPart(char)
    if not aim then setLockedTarget(nil); return end

    -- wall check
    if WallCheckToggle.Value then
        local origin = Camera.CFrame.Position
        local dir = aim.Position - origin
        local rp = RaycastParams.new()
        rp.FilterDescendantsInstances = {LocalPlayer.Character}
        rp.FilterType = Enum.RaycastFilterType.Blacklist
        rp.IgnoreWater = true
        local rc = Workspace:Raycast(origin, dir, rp)
        if rc and rc.Instance and not rc.Instance:IsDescendantOf(char) then return end
    end

    -- prediction + jump offset + smoothness
    local predictedPos, smooth = calculateTargetPos(aim)
    local alpha = math.clamp(smooth * dt * 60, 0, 1)
    Camera.CFrame = Camera.CFrame:Lerp(CFrame.new(Camera.CFrame.Position, predictedPos), alpha)

    -- dot
    if DotToggle.Value then
        local bp, anchor = ensureDotOnCharacter(char)
        if bp and anchor then
            local dotframe = bp:FindFirstChild("dotframe")
            if dotframe then
                local ds = ton(DotSizeSlider.Value, DEFAULTS.DotSize)
                if bp.Size.X.Offset ~= ds then bp.Size = UDim2.new(0, ds, 0, ds) end
                local t = tick()
                local hue = (t * 0.15) % 1
                dotframe.BackgroundColor3 = Color3.fromHSV(hue,0.9,1)
            end
        end
    else
        cleanupDotFromCharacter(char)
    end
end)
