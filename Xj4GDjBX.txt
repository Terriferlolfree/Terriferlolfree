local function SetDraggable(self: GuiObject)
    table.insert(Draggables, self)
    local DragOrigin
    local GuiOrigin
    local ActiveInputType

    local function OnInputBegan(Input: InputObject)
        if Dragging.Gui or Dragging.True then return end
        if Input.UserInputType ~= Enum.UserInputType.MouseButton1 and
           Input.UserInputType ~= Enum.UserInputType.Touch then return end

        for _, v in ipairs(Draggables) do
            v.ZIndex = 1
        end
        self.ZIndex = 2

        ActiveInputType = Input.UserInputType
        Dragging = {Gui = self, True = true}
        DragOrigin = Vector2.new(Input.Position.X, Input.Position.Y)
        GuiOrigin = self.Position
    end

    local function OnInputEnded(Input: InputObject)
        if Input.UserInputType == ActiveInputType then
            Dragging = {Gui = nil, True = false}
            ActiveInputType = nil
        end
    end

    local function OnInputChanged(Input: InputObject)
        if Dragging.Gui ~= self then return end
        if Input.UserInputType ~= Enum.UserInputType.MouseMovement and
           Input.UserInputType ~= Enum.UserInputType.Touch then return end

        local Delta = Vector2.new(Input.Position.X, Input.Position.Y) - DragOrigin
        local ScreenSize = Menu.ScreenSize

        local ScaleX = ScreenSize.X * GuiOrigin.X.Scale
        local ScaleY = ScreenSize.Y * GuiOrigin.Y.Scale
        local OffsetX = GuiOrigin.X.Offset + Delta.X + ScaleX
        local OffsetY = GuiOrigin.Y.Offset + Delta.Y + ScaleY

        self.Position = UDim2.fromOffset(OffsetX, OffsetY)
    end

    self.InputBegan:Connect(OnInputBegan)
    self.InputEnded:Connect(OnInputEnded)
    UserInput.InputChanged:Connect(OnInputChanged)
end

        if Item.Tab == Tab_Name and Item.Container == Container_Name then
            if Item.Name == Name and (Item.Class == Class_Name) then

function Menu.Container(Tab_Name: string, Container_Name: string, Side: string): Container
    local Tab = GetTab(Tab_Name)
    assert(typeof(Tab_Name) == "string", "TAB_NAME REQUIRED")
    if Tab[Container_Name] then return error("CONTAINER_NAME '" .. tostring(Container_Name) .. "' ALREADY EXISTS") end
    local Side = Side or "Left"

    local Frame = Instance.new("Frame")

    local Stroke = Instance.new("Frame")
    Stroke.Name = "Stroke"
    Stroke.BackgroundColor3 = Menu.Accent
    Stroke.Size = UDim2.new(1, 0, 0, 2)
    Stroke.Position = UDim2.new(0, 0, 0, 0)
    Stroke.BorderSizePixel = 0
    Stroke.Parent = Frame

    local Label = CreateLabel(Frame, "Title", Container_Name, UDim2.fromOffset(206, 15), UDim2.fromOffset(5, 2))

    local Container = {self = Frame, Height = 0}
    Container.Class = "Container"
    Container.Visible = true

    function Container:SetLabel(Name: string)
        Label.Text = tostring(Name)
    end

    function Container:SetVisible(Visible: boolean)
        if typeof(Visible) ~= "boolean" then return end
        if self.Visible == Visible then return end
        
        Frame.Visible = Visible
        self.Visible = Visible
        self:UpdateSize(Visible and 25 or -25, Frame)
    end

    function Container:UpdateSize(Height: float, Item: GuiObject)
        self.Height += Height
        Frame.Size += UDim2.fromOffset(0, Height)
        Tab.self[Side].CanvasSize += UDim2.fromOffset(0, Height)

        if Item then
            local ItemY = Item.AbsolutePosition.Y
            if math.sign(Height) == 1 then
                ItemY -= 1
            end

            for _, item in ipairs(Frame:GetChildren()) do
                if (item == Label or item == Stroke or Item == item) then continue end -- exclude these
                local item_y = item.AbsolutePosition.Y
                if item_y > ItemY then
                    item.Position += UDim2.fromOffset(0, Height)
                end
            end
        end
    end

    function Container:GetHeight(): number

function Menu.Slider(Tab_Name: string, Container_Name: string, Name: string, Min: number, Max: number, Value: number, Unit: string, Scale: number, Callback: any, ToolTip: string): Slider
    local Container = GetContainer(Tab_Name, Container_Name)
    local Label = CreateLabel(Container.self, "Slider", Name, UDim2.new(1, -10, 0, 15), UDim2.fromOffset(20, Container:GetHeight()))
    local Button = Instance.new("TextButton")
    local ValueBar = Instance.new("TextLabel")
    local ValueBox = Instance.new("TextBox")
    local ValueLabel = Instance.new("TextLabel")

    local ValueGlow = Instance.new("ImageLabel")
    ValueGlow.Image = "http://www.roblox.com/asset/?id=18245826428"
    ValueGlow.ScaleType = Enum.ScaleType.Slice
    ValueGlow.ImageTransparency = 0.55
    ValueGlow.ImageColor3 = Menu.Accent
    ValueGlow.BackgroundTransparency = 1
    ValueGlow.ZIndex = 2
    ValueGlow.SliceCenter = Rect.new(Vector2.new(21, 21), Vector2.new(79, 79))
    ValueGlow.Size = UDim2.new(1, 8, 1, 8)
    ValueGlow.Position = UDim2.new(0, -4, 0, -4)
    ValueGlow.Visible = false
    ValueGlow.Parent = ValueBar

    local Slider = {}
    Slider.Name = Name
    Slider.Class = "Slider"
    Slider.Tab = Tab_Name
    Slider.Container = Container_Name
    Slider.Index = #Items + 1
    Slider.Min = typeof(Min) == "number" and math.clamp(Min, Min, Max) or 0
    Slider.Max = typeof(Max) == "number" and Max or 100
    Slider.Value = typeof(Value) == "number" and Value or 100
    Slider.Unit = typeof(Unit) == "string" and Unit or ""
    Slider.Scale = typeof(Scale) == "number" and Scale or 0
    Slider.Callback = typeof(Callback) == "function" and Callback or function() end


    local function UpdateSlider(Percentage: number)
        local Percentage = typeof(Percentage) == "number" and math.clamp(Percentage, 0, 1) or 0
        local Value = Slider.Min + ((Slider.Max - Slider.Min) * Percentage)
        local Scale = (10 ^ Slider.Scale)
        Slider.Value = math.round(Value * Scale) / Scale

        ValueBar.Size = UDim2.new(Percentage, 0, 0, 5)
        ValueBox.Text = "[" .. Slider.Value .. "]"
        ValueLabel.Text = Slider.Value .. Slider.Unit
        
        -- Show glow when there's value and hide when at minimum
        ValueGlow.Visible = Percentage > 0
    end


    function Slider:Update(Percentage: number)
        UpdateSlider(Percentage)
    end

    function Slider:SetLabel(Name: string)
        Label.Text = tostring(Name)
    end

    function Slider:SetVisible(Visible: boolean)
        if typeof(Visible) ~= "boolean" then return end
        if Label.Visible == Visible then return end
        
        Label.Visible = Visible
        Container:UpdateSize(Visible and 30 or -30, Label)
    end

    function Slider:GetValue(): number
        return self.Value
    end

    function Slider:SetValue(Value: number)
        self.Value = typeof(Value) == "number" and math.clamp(Value, self.Min, self.Max) or self.Min
        local Percentage = (self.Value - self.Min) / (self.Max - self.Min)
        self:Update(Percentage)
    end

    Slider.self = Label

    Label.MouseEnter:Connect(function()
        if ToolTip then
            Menu:SetToolTip(true, ToolTip, Label)
        end
    end)
    Label.MouseLeave:Connect(function()
        Menu:SetToolTip(false)
    end)

    Button.Name = "Slider"
    Button.BackgroundColor3 = Color3.fromRGB(15, 15, 15)
    Button.BorderColor3 = Color3.new()
    Button.Position = UDim2.fromOffset(0, 20)
    Button.Size = UDim2.new(1, -40, 0, 5)
    Button.Text = ""
    Button.AutoButtonColor = false
    Button.Parent = Label
    CreateCorner(Button, 3)
    CreateStroke(Button, Color3.new(), 1)

    ValueBar.Name = "ValueBar"
    ValueBar.BackgroundColor3 = Menu.Accent
    ValueBar.BorderSizePixel = 0
    ValueBar.Size = UDim2.fromScale(1, 1)
    ValueBar.Text = ""
    ValueBar.Parent = Button
    AddEventListener(ValueBar, function()
        ValueBar.BackgroundColor3 = Menu.Accent
    end)
    CreateCorner(ValueBar, 3)
    
    ValueBox.Name = "ValueBox"
    ValueBox.BackgroundTransparency = 1
    ValueBox.Position = UDim2.new(1, -65, 0, 0)
    ValueBox.Size = UDim2.fromOffset(50, 10)
    ValueBox.FontFace = Settings.Font
    ValueBox.Text = ""
    ValueBox.TextColor3 = Color3.new(1, 1, 1)
    ValueBox.TextSize = 12
    ValueBox.TextXAlignment = Enum.TextXAlignment.Right
    ValueBox.ClipsDescendants = true
    ValueBox.Parent = Label
    ValueBox.FocusLost:Connect(function()
        Slider.Value = tonumber(ValueBox.Text) or 0
        local Percentage = (Slider.Value - Slider.Min) / (Slider.Max - Slider.Min)
        Slider:Update(Percentage)
        Slider.Callback(Slider.Value)
    end)

    ValueLabel.Name = "ValueLabel"
    ValueLabel.BackgroundTransparency = 1
    ValueLabel.Position = UDim2.new(1, 0, 0, 2)
    ValueLabel.Size = UDim2.new(0, 0, 1, 0)
    ValueLabel.FontFace = Settings.Font
    ValueLabel.Text = ""
    ValueLabel.TextSize = 14
    ValueLabel.Parent = ValueBar
    ValueLabel.Transparency = 1

    Button.InputBegan:Connect(function(Input: InputObject, Process: boolean)
        if (not Dragging.Gui and not Dragging.True) and (Input.UserInputType == Enum.UserInputType.MouseButton1) then
            Dragging = {Gui = Button, True = true}
            local InputPosition = Vector2.new(Input.Position.X, Input.Position.Y)
            local Percentage = (InputPosition - Button.AbsolutePosition) / Button.AbsoluteSize
            Slider:Update(Percentage.X)
            Slider.Callback(Slider.Value)
        end
    end)

    UserInput.InputChanged:Connect(function(Input: InputObject, Process: boolean)
        if Dragging.Gui ~= Button then return end
        if not (UserInput:IsMouseButtonPressed(Enum.UserInputType.MouseButton1)) then
            Dragging = {Gui = nil, True = false}
            return
        end
        if (Input.UserInputType == Enum.UserInputType.MouseMovement) then
            local InputPosition = Vector2.new(Input.Position.X, Input.Position.Y)
            local Percentage = (InputPosition - Button.AbsolutePosition) / Button.AbsoluteSize
            Slider:Update(Percentage.X)
            Slider.Callback(Slider.Value)
        end
    end)


    Slider:SetValue(Slider.Value)
    Container:UpdateSize(30)
    table.insert(Items, Slider)
    return #Items
end

                itemCount += 1
            end
            if itemCount >= 9 then -- Will be 10 after this item
                List.CanvasSize += UDim2.fromOffset(0, 15)
            end
        end
        table.insert(ItemObjects, Button)
    end

    function ListBox:Update(Value: string, Items: any)
        if self.Method == "Default" then
            if Value then
                UpdateValue(Value)
            end
        end
        if typeof(Items) == "table" then
            if self.Method == "Multi" then
                self.Items = Items -- Fixed: Use Items parameter, not Value
                UpdateValue()
            else
                self.Items = Items
            end
            
            -- Clear existing items
            for _, Button in ipairs(ItemObjects) do
                Button:Destroy()
            end
            table.clear(ItemObjects)

            List.CanvasSize = UDim2.new()
            List.Size = UDim2.new(1, -50, 0, 150)
            
            if self.Method == "Default" then
                for _, Item in ipairs(self.Items) do
                    if typeof(Item) == "table" and Item.Name and Item.DisplayText then
                        AddItem(Item.Name, false, Item.DisplayText)
                    else
                        AddItem(tostring(Item), false)
                    end
                end
            else
                for Name, Checked in pairs(self.Items) do
                    -- Handle both old boolean format and new table format
                    if typeof(Checked) == "table" then
                        if Checked.Selected ~= nil and Checked.DisplayText then
                            AddItem(Name, Checked.Selected, Checked.DisplayText)
                        else
                            AddItem(Name, false)
                        end
                    else
                        AddItem(Name, Checked)
                    end
                end
            end
        else
            -- Update existing items without recreating them
            if self.Method == "Default" then
                for _, Button in ipairs(ItemObjects) do
                    Button.TextColor3 = self.Value == Button.Text and Menu.Accent or Color3.new(1, 1, 1)
                end
            else
                -- Update visual state for multi-select without recreating
                UpdateValue() -- Refresh the value first
                for _, Button in ipairs(ItemObjects) do
                    -- Find the corresponding item by matching the display text or name
                    local buttonName = nil
                    local buttonText = Button.Text
                    
                    -- Try to extract the username from display text (before " | @")
                    local extractedName = buttonText:match("^([^%s|]+)")
                    if extractedName and self.Items[extractedName] then
                        buttonName = extractedName
                    else
                        -- Fallback: check if button text matches any item key directly
                        for itemName, _ in pairs(self.Items) do
                            if itemName == buttonText then
                                buttonName = itemName
                                break
                            end
                        end
                    end
                    
                    if buttonName then
                        local itemData = self.Items[buttonName]
                        if typeof(itemData) == "table" then
                            Button.TextColor3 = itemData.Selected and Menu.Accent or Color3.new(1, 1, 1)
                        else
                            Button.TextColor3 = itemData and Menu.Accent or Color3.new(1, 1, 1)
                        end
                    end
                end
            end
        end
    end

    function ListBox:SetVisible(Visible: boolean)
        if typeof(Visible) ~= "boolean" then return end
        if List.Visible == Visible then return end
        
        List.Visible = Visible
        Container:UpdateSize(Visible and 155 or -155, List)
    end

    function ListBox:SetValue(Value: string, Items: any)
        if self.Method == "Default" then
            if typeof(Items) == "table" then
                self.Items = Items
            end
            self:Update(Value, self.Items)
        else
            if typeof(Value) == "table" then
                self.Items = Value
            end
            self:Update(nil, self.Items)
        end
    end

    function ListBox:GetValue(): table
        return self.Value
    end

    -- Initialize List properties
    List.Name = "List"
    List.Active = true
    List.BackgroundColor3 = Menu.ItemColor
    List.BorderColor3 = Color3.new()
    List.Position = UDim2.fromOffset(20, Container:GetHeight())
    List.Size = UDim2.new(1, -50, 0, 150)
    List.CanvasSize = UDim2.new()
    List.ScrollBarThickness = 4
    List.ScrollBarImageColor3 = Menu.Accent
    List.Parent = Container.self
    List.MouseEnter:Connect(function()
        if ToolTip and ToolTip ~= "" then
            Menu:SetToolTip(true, ToolTip, List)
        end
    end)
    List.MouseLeave:Connect(function()
        if ToolTip and ToolTip ~= "" then
            Menu:SetToolTip(false)
        end
    end)
    CreateStroke(List, Color3.new(), 1)
    AddEventListener(List, function()
        List.BackgroundColor3 = Menu.ItemColor
        List.ScrollBarImageColor3 = Menu.Accent
    end)

    ListLayout.Parent = List

    -- Initialize the ListBox
    if ListBox.Method == "Default" then
        ListBox:Update(ListBox.Value, ListBox.Items)
    else
        ListBox:Update(nil, ListBox.Items)
    end
    
    Container:UpdateSize(155)
    table.insert(Items, ListBox)
    return #Items
end

function Menu:SaveConfig(Name)
    local data = {}

    for _, Item in ipairs(Items) do
        if not Savable[Item.Class] then
            continue
        end

        if not Item.GetValue then
            continue
        end

        local shouldSave = false
        local valueToSave = nil

        if Item.Class == "ColorPicker" then
            local col, alpha = Item:GetValue()
            -- Save if it's not the default color
            if col.R ~= 1 or col.G ~= 1 or col.B ~= 1 or alpha ~= 1 then
                shouldSave = true
                valueToSave = {col.R, col.G, col.B, alpha}
            end

        elseif Item.Class == "Hotkey" then
            local key, mode = Item:GetValue()
            -- Only save if a key is actually set
            if key then
                shouldSave = true
                valueToSave = {key.Name, mode}
            end

        elseif Item.Class == "CheckBox" then
            local value = Item:GetValue()
            -- Only save if checkbox is enabled (true)
            if value == true then
                shouldSave = true
                valueToSave = value
            end

        elseif Item.Class == "Slider" then
            local value = Item:GetValue()
            -- Always save slider values (even if 0)
            shouldSave = true
            valueToSave = value

        elseif Item.Class == "TextBox" then
            local value = Item:GetValue()
            -- Only save if textbox has content
            if value and value ~= "" then
                shouldSave = true
                valueToSave = value
            end

        elseif Item.Class == "ComboBox" then
            local value = Item:GetValue()
            -- Only save if something is selected (string check)
            if value and type(value) == "string" and value ~= "" then
                shouldSave = true
                valueToSave = value
            end

        elseif Item.Class == "MultiSelect" then
            local value = Item:GetValue()
            -- Check if any item is selected (dictionary format)
            local hasSelection = false
            if value and type(value) == "table" then
                for _, selected in pairs(value) do
                    if selected == true then
                        hasSelection = true
                        break
                    end
                end
            end
            if hasSelection then
                shouldSave = true
                valueToSave = value
            end

        end

        -- Only add to config if it should be saved
        if shouldSave then
            local tab = Item.Tab
            local container = Item.Container
            local name = Item.Name

            data[tab] = data[tab] or {}
            data[tab][container] = data[tab][container] or {}
            data[tab][container][name] = valueToSave
        end
    end

    writefile(
        ConfigFolder .. "/" .. Name .. ".json",
        HttpService:JSONEncode(data)
    )

    Menu.sendnoti("Config Saved", "Saved as <font color='#00ff00'>" .. Name .. "</font>", 3)
end
function Menu:LoadConfig(Name)
    local path = ConfigFolder .. "/" .. Name .. ".json"
    if not isfile(path) then
        Menu.sendnoti("Error", "Config not found", 3)
        return
    end

    local data = HttpService:JSONDecode(readfile(path))

    for _, Item in ipairs(Items) do
        if not Savable[Item.Class] then continue end
        if not Item.SetValue then continue end

        local value =
            data[Item.Tab] and
            data[Item.Tab][Item.Container] and
            data[Item.Tab][Item.Container][Item.Name]

        if value == nil then continue end

        if Item.Class == "ColorPicker" then
            local col = Color3.new(value[1], value[2], value[3])
            Item:SetValue(col, value[4])
            if Item.Callback then Item.Callback(col, value[4]) end

        elseif Item.Class == "Hotkey" then
            local key = value[1] ~= "None" and Enum.KeyCode[value[1]] or nil
            Item:SetValue(key, value[2])
            if Item.Callback then Item.Callback(key, value[2]) end

        elseif Item.Class == "Slider" then
            local num = tonumber(value)
            if num then
                Item:SetValue(num)
                if Item.Callback then Item.Callback(Item.Value) end
            end

        else
            Item:SetValue(value)
            if Item.Callback then Item.Callback(value) end
        end
    end

    Menu.sendnoti("Success", "Config loaded successfully", 3)
end
function Menu:GetConfigs()
    local list = {}

    for _, file in ipairs(listfiles(ConfigFolder)) do
        local name = file:match("([^/\\]+)%.json$")
        if name then
            table.insert(list, name)
        end
    end

    return list
end

function addVelHistory(vel)
    table.insert(velHistory, vel)
    if #velHistory > maxVelHistory then
        table.remove(velHistory, 1)
    end
end
function getSmoothedVel()
    if #velHistory == 0 then return Vector3.new() end
    local sum = Vector3.new()
    for i = 1, #velHistory do
        sum = sum + velHistory[i]
    end
    return sum / #velHistory
end
function ZetaGetClosest()
    local c,d = nil,1/0
    local m = game:GetService("UserInputService"):GetMouseLocation()
    for _,pl in ipairs(PlayersSvc:GetPlayers()) do
        if pl~=Me and pl.Character and pl.Character:FindFirstChild("HumanoidRootPart") then
            local root = pl.Character.HumanoidRootPart
            local hum = pl.Character:FindFirstChild("Humanoid")
            if hum and hum.Health > 0 then
                local pos,ons = Cam:WorldToViewportPoint(root.Position)
                if ons then
                    local dist = (Vector2.new(pos.X,pos.Y)-m).Magnitude
                    if dist<d then d=dist c=pl end
                end
            end
        end
    end
    return c
end
function ZetaEnable()
    ZetaDisable()
    ZetaInputConn = game:GetService("UserInputService").InputBegan:Connect(function(i,gp)
        if not gp and i.KeyCode==ZetaKey then
            if not ZetaEnabled then
                ZetaTarget = ZetaGetClosest()
                ZetaEnabled = ZetaTarget~=nil
                velHistory = {}
            else
                ZetaTarget = nil
                ZetaEnabled = false
                velHistory = {}
            end
        end
    end)
    ZetaConn = RunSrv.RenderStepped:Connect(function(dt)
        if not (ZetaEnabled and ZetaTarget and ZetaTarget.Character) then return end
        local tc = ZetaTarget.Character
        local tp = tc:FindFirstChild(AimAssistTargetPart) or tc:FindFirstChild("HumanoidRootPart")
        if not tp then return end
        
        local hum = tc:FindFirstChild("Humanoid")
        if hum and hum.Health <= 0 then
            ZetaTarget = nil
            ZetaEnabled = false
            return
        end
        
        -- Knock check: unlock if health is 1 or 2
        if ZetaKnockCheck and hum and hum.Health <= 2 then
            ZetaTarget = nil
            ZetaEnabled = false
            velHistory = {}
            return
        end
        
        local camPos = Cam.CFrame.Position
        local tgtPos = tp.Position
        local hrp = tc:FindFirstChild("HumanoidRootPart")
        local vel = (hrp and hrp.Velocity) or Vector3.new()
        
        -- Smooth velocity over frames
        addVelHistory(vel)
        local smoothVel = getSmoothedVel()
        
        -- Enhanced prediction with distance scaling
        local distance = (tgtPos - camPos).Magnitude
        local distScale = math.clamp(distance / 100, 0.7, 2)
        local lead = smoothVel * ZetaLeadFactor * distScale
        
        local desiredVec = (tgtPos+lead)-camPos
        if desiredVec.Magnitude<=0 then return end
        local desiredDir = desiredVec.Unit
        local currDir = Cam.CFrame.LookVector
        local dot = math.clamp(currDir:Dot(desiredDir),-1,1)
        local angle = math.acos(dot)
        local maxAngleThisFrame = ZetaMaxAngularSpeed*math.max(dt,1/120)
        if angle>1e-4 then
            local t = math.min(1,maxAngleThisFrame/angle)
            local intermediate = (currDir*(1-t)+desiredDir*t)
            if intermediate.Magnitude==0 then return end
            intermediate = intermediate.Unit
            local up = Vector3.new(0,1,0)
            local perp = intermediate:Cross(up)
            if perp.Magnitude<0.001 then perp = intermediate:Cross(Vector3.new(1,0,0)) end
            perp = perp.Unit
            local jitter = (math.random()-0.5)*2*ZetaJitter*math.clamp(desiredVec.Magnitude/200,0.5,1.5)
            intermediate = (intermediate+perp*jitter).Unit
            local newC = CFrame.new(camPos,camPos+intermediate)
            local alpha = math.clamp(1-math.exp(-ZetaSmooth*60*dt),0,1)
            Cam.CFrame = Cam.CFrame:Lerp(newC,alpha)
        end
    end)
end

    if ZetaInputConn then ZetaInputConn:Disconnect() ZetaInputConn=nil end
    ZetaEnabled = false
    ZetaTarget = nil
    velHistory = {}
end

function isKO(p)
	return p and p.Parent and p.Character and p.Character:FindFirstChild("BodyEffects") 
		and p.Character.BodyEffects:FindFirstChild("K.O")

function isFullyLoaded(player)
	if not player or not player.Character then return false end
	
	local char = player.Character

	local humanoid = char:FindFirstChildOfClass("Humanoid")
	local rootPart = char:FindFirstChild("HumanoidRootPart")
	local bodyEffects = char:FindFirstChild("BodyEffects")
	
	if not humanoid or not rootPart or not bodyEffects then
		return false
	end

	local hasForceField = char:FindFirstChild("ForceField") or char:FindFirstChild("ForceField_TESTING")
    local isfullyloaded = char:FindFirstChild("FULLY_LOADED_CHAR")
	
	if hasForceField and not isfullyloaded then
		return false
	end

	local koValue = bodyEffects:FindFirstChild("K.O")
	if not koValue then
		return false
	end
	
	return true
end
function getNearestPlayerToMouse()
    local mouse = local_player:GetMouse()
    if not mouse then return nil end
    local cam = workspace.CurrentCamera
    if not cam then return nil end
    local nearest = nil
    local shortestDistance = math.huge

    for _, player in pairs(plrs:GetPlayers()) do
        if player == local_player then continue end

        local char = player.Character
        if not char then continue end

        if targetAimConditions and targetAimConditions["Alive Check"] then
            if isKO and isKO(player) then continue end
        end

        if targetAimConditions and targetAimConditions["Force Field Check"] then
            if isFullyLoaded and not isFullyLoaded(player) then continue end
        end

        local head = char:FindFirstChild("Head")
        if not head then continue end

        local screenPos, onScreen = cam:WorldToViewportPoint(head.Position)
        if not onScreen then continue end

        local mousePos = Vector2.new(mouse.X, mouse.Y)
        local headPos = Vector2.new(screenPos.X, screenPos.Y)
        local distance = (mousePos - headPos).Magnitude

        if distance < shortestDistance then
            shortestDistance = distance
            nearest = player
        end
    end

    return nearest
end

function vpGenerateDeepVoid()
    local pos = {}
    local deep = {-2e6, -5e6, -1e7, -2e7, -5e7}
    local far  = {-1e6, -5e5, -25e4, -1e5, 1e5, 25e4, 5e5, 1e6}

    for i = 1, 20 do
        local x, y, z
        local d = deep[math.random(1, #deep)]
        local f1 = far[math.random(1, #far)]
        local f2 = far[math.random(1, #far)]
        local r = math.random(-1e4, 1e4)

        if vpDirection == "-Y" then
            x = f1 + r y = d + r z = f2 + r
        elseif vpDirection == "+Y" then
            x = f1 + r y = -d + r z = f2 + r
        elseif vpDirection == "-X" then
            x = d + r y = f1 + r z = f2 + r
        elseif vpDirection == "+X" then
            x = -d + r y = f1 + r z = f2 + r
        elseif vpDirection == "-Z" then
            x = f1 + r y = f2 + r z = d + r
        elseif vpDirection == "+Z" then
            x = f1 + r y = f2 + r z = -d + r
        end

        table.insert(pos, Vector3.new(x, y, z))
    end

    return pos
end

function vpGenerateBreakNull()
    local pos = {}
    local bv  = 1e16
    local far = {-1e8, -5e7, -25e6, -1e7, 1e7, 25e6, 5e7, 1e8}

    for i = 1, 20 do
        local x, y, z
        local f1 = far[math.random(1, #far)]
        local f2 = far[math.random(1, #far)]
        local r  = math.random(-1e4, 1e4)

        if vpDirection == "-Y" then
            x = f1 + r y = -bv z = f2 + r
        elseif vpDirection == "+Y" then
            x = f1 + r y = bv  z = f2 + r
        elseif vpDirection == "-X" then
            x = -bv    y = f1 + r z = f2 + r
        elseif vpDirection == "+X" then
            x = bv     y = f1 + r z = f2 + r
        elseif vpDirection == "-Z" then
            x = f1 + r y = f2 + r z = -bv
        elseif vpDirection == "+Z" then
            x = f1 + r y = f2 + r z = bv
        end

    end

    return pos
end

            if rl then rl.CFrame = lookAtCFrame * CFrame.new( 0.5, -2,   0) end
        end

        shared.headRef = th
        shared.ready = true

                    ME:FireServer("ShootGun", handle, handle.Position, futureHeadPos, th, Vector3.new(0,0,0))
                end)
            end
        end
    end)

    task.spawn(function()
        while isActive do
            if not targetAimActive then performCleanup() break end
            task.wait(0.02)
        end
    end)

            continue
        end

        if not c then
            getgenv().selfLastHealth = nil
            continue
        end

            getgenv().selfLastHealth = nil
            continue
        end

        local humanoid    = c:FindFirstChildOfClass("Humanoid")
        local bodyEffects = c:FindFirstChild("BodyEffects")

        if not humanoid or not bodyEffects then continue end

        local healthNow = humanoid.Health

        if getgenv().selfLastHealth == nil then
            getgenv().selfLastHealth = healthNow
            continue
        end

        if healthNow > getgenv().selfLastHealth + 50 then
            getgenv().selfLastHealth = healthNow
            continue
        end

        if healthNow + 0.05 < getgenv().selfLastHealth then
            getgenv().selfLastHealth = healthNow
            task.wait(0.1)

            if not local_player.Character or local_player.Character ~= c then
                getgenv().selfLastHealth = nil
                continue
            end

            local recheck      = c:FindFirstChild("BodyEffects")
            local lastDamager  = recheck and recheck:FindFirstChild("LastDamager")
            local attackerName = lastDamager and tostring(lastDamager.Value)

            if not attackerName or attackerName == "" then continue end

            local attacker = Players:FindFirstChild(attackerName)
            if not attacker or attacker == local_player then continue end
            if not attacker.Parent or not attacker.Character then continue end
            if targetAimConditions["Alive Check"] and isKO(attacker) then continue end
            if targetAimConditions["Force Field Check"] and tc:FindFirstChildOfClass("ForceField") then
                Menu.sendnoti("Defense", "Waiting for target to load...", 1.5)
                continue
            end

            if targetAimActive then continue end

            if returnToPosition and not originalSavedPosition then
                local fc = local_player.Character
                if fc and fc:FindFirstChild("HumanoidRootPart") then
                    originalSavedPosition = fc.HumanoidRootPart.CFrame
                end
            end

            savedPosition   = nil
            currentTarget   = attacker
            targetAimActive = true
            Menu.sendnoti("Defense", attacker.Name.." Has damage you", 1.5)
            targetAimKnockCFrame()
        else
            getgenv().selfLastHealth = healthNow
        end
    end
end)

local function has_item(name)
    local c = get_char()
    if not c then return false end
    return c:FindFirstChild(name) ~= nil
        or local_player.Backpack:FindFirstChild(name) ~= nil
end

local function get_selected_guns()
    local t = {}
    for name, checked in pairs(SelectedGuns) do
        if checked then table.insert(t, name) end
    end
    return t

local function do_buy(shopName, pre, post)
    task.spawn(function()
        local hrp = get_hrp()
        if not hrp then shop_busy = false return end

        local shopItem = game.Workspace.Ignored.Shop:FindFirstChild(shopName)
        if not shopItem or not shopItem:FindFirstChild("Head") then
            shop_busy = false return
        end

        if pre then pcall(pre) end

        SavedPosition = hrp.CFrame

        hrp.CFrame = shopItem.Head.CFrame

        task.wait(0.2)

        pcall(function()
            fireclickdetector(shopItem:FindFirstChildOfClass("ClickDetector"))
        end)

        task.wait(0.3)

        hrp.CFrame = SavedPosition

        task.wait(0.1)

        if post then pcall(post) end

        shop_busy = false
    end)
end

local function queue_buy(shopName, pre, post)
    table.insert(shop_queue, { shopName = shopName, pre = pre, post = post })
end

local function process_queue()
    if shop_busy or #shop_queue == 0 then return end
    shop_busy   = true
    local entry = table.remove(shop_queue, 1)
    do_buy(entry.shopName, entry.pre, entry.post)
end

local function stopFly()

local function doFly()
    if flyMethod == "Camera Direction" then
        local c = lp.Character
        if not c then return end
        local root = c:FindFirstChild("HumanoidRootPart")
        local hum  = c:FindFirstChildOfClass("Humanoid")
        if not root or not hum then return end

        hum.PlatformStand = true

        if workspace:FindFirstChild("FlyCore") then workspace.FlyCore:Destroy() end
        local Core        = Instance.new("Part")
        Core.Name         = "FlyCore"
        Core.Size         = Vector3.new(0.05, 0.05, 0.05)
        Core.CanCollide   = false
        Core.Transparency = 1
        Core.Parent       = workspace

        local Weld = Instance.new("Weld", Core)
        Weld.Part0 = Core
        Weld.Part1 = root
        Weld.C0    = CFrame.new(0, 0, 0)

        local BV     = Instance.new("BodyVelocity", Core)
        BV.MaxForce  = Vector3.new(9e9, 9e9, 9e9)
        BV.Velocity  = Vector3.zero

        local BG     = Instance.new("BodyGyro", Core)
        BG.MaxTorque = Vector3.new(9e9, 9e9, 9e9)
        BG.P         = 9e4
        BG.CFrame    = Core.CFrame

        spawn(function()
            while fly do
                if UserInputService:GetFocusedTextBox() then task.wait() continue end
                local cam = workspace.CurrentCamera
                local dir = Vector3.zero
                if UserInputService:IsKeyDown(Enum.KeyCode.W)           then dir += cam.CFrame.LookVector  end
                if UserInputService:IsKeyDown(Enum.KeyCode.S)           then dir -= cam.CFrame.LookVector  end
                if UserInputService:IsKeyDown(Enum.KeyCode.A)           then dir -= cam.CFrame.RightVector end
                if UserInputService:IsKeyDown(Enum.KeyCode.D)           then dir += cam.CFrame.RightVector end
                if UserInputService:IsKeyDown(Enum.KeyCode.Space)       then dir += Vector3.new(0, 1, 0)   end
                if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then dir -= Vector3.new(0, 1, 0)   end
                BV.Velocity = dir.Magnitude > 0 and dir.Unit * flyspd or Vector3.zero
                BG.CFrame   = cam.CFrame
                task.wait()
            end
            if workspace:FindFirstChild("FlyCore") then workspace.FlyCore:Destroy() end
            if hum then hum.PlatformStand = false end
        end)

    elseif flyMethod == "Keyboard Direction" then
        spawn(function()
            while fly do
                local c = lp.Character
                if not c then break end
                local root = c:FindFirstChild("HumanoidRootPart")
                if not root then break end
                local cam = workspace.CurrentCamera
                local dir = Vector3.zero
                if UserInputService:IsKeyDown(Enum.KeyCode.W) then dir = dir + cam.CFrame.LookVector  end
                if UserInputService:IsKeyDown(Enum.KeyCode.S) then dir = dir - cam.CFrame.LookVector  end
                if UserInputService:IsKeyDown(Enum.KeyCode.A) then dir = dir - cam.CFrame.RightVector end
                if UserInputService:IsKeyDown(Enum.KeyCode.D) then dir = dir + cam.CFrame.RightVector end
                root.AssemblyLinearVelocity = dir.Magnitude > 0 and dir.Unit * flyspd or Vector3.new()
                task.wait()
            end
            local c = lp.Character
            if c and c:FindFirstChild("HumanoidRootPart") then
                c.HumanoidRootPart.AssemblyLinearVelocity = Vector3.new()
            end
        end)
    end
end

    rifle     = CFrame.new(0, 0, 0),
    aug       = CFrame.new(0, 0, 0) * CFrame.Angles(0, math.rad(90), 0),
    flintlock = CFrame.new(0, 0, 0) * CFrame.Angles(0, math.rad(90), 0),

function SC_GetTool(gunName)
    local c  = lp.Character
    local bp = lp.Backpack
    if c then
        for _, t in ipairs(c:GetChildren()) do
            if t:IsA("Tool") and t.Name:lower():sub(2,-2) == gunName then return t end
        end
    end
    if bp then
        for _, t in ipairs(bp:GetChildren()) do
            if t:IsA("Tool") and t.Name:lower():sub(2,-2) == gunName then return t end
        end
    end
end

function SC_Clean(default, handle)
    for _, child in ipairs(default:GetChildren()) do
        if #child.Name == 0 then child:Destroy() end
    end
    if handle then
        for _, child in ipairs(handle:GetChildren()) do
            if #child.Name == 0 then child:Destroy() end
        end
    end
end

function SC_Apply(gunName, skinName)
    local tool = SC_GetTool(gunName)
    if not tool then return end

    local handle  = tool:FindFirstChild("Handle")
    local default = tool:FindFirstChild("Default")
    if not default then return end

    local skinData = SC_defaults[gunName]

    if not SC_originalData[gunName] then
        SC_originalData[gunName] = {
            TextureID    = default.TextureID,
            Transparency = default.Transparency,
            Massless     = default.Massless,
            CanCollide   = default.CanCollide
        }
    end

    handle:SetAttribute("SkinName", skinName)
    SC_Clean(default, handle)

    default.Massless   = true
    default.CanCollide = false

    local mesh = SC_meshes[skinName]

    if mesh then
        local newMesh = mesh.ClassName == "MeshPart" and mesh or nil

        if not newMesh then
            for _, child in pairs(mesh) do
                if child.Name:lower():find(gunName:lower()) then
                    newMesh = child
                    break
                end
            end
        end

        if newMesh then
            local fake       = newMesh:Clone()
            fake.CFrame      = default.CFrame
            fake.Name        = "\0"
            fake.CanCollide  = false
            fake.Anchored    = false
            fake.Massless    = true
            fake.CastShadow  = false
            fake.Parent      = default

            local weld  = Instance.new("Weld")
            weld.Part0  = default
            weld.Part1  = fake
            weld.C0     = SC_offsets[gunName] or CFrame.new()
            weld.Name   = "\0"
            weld.Parent = default

            default.Transparency = 1
        end
    else
        default.Transparency = 0
        if skinData and skinData.TextureID then
            default.TextureID = typeof(skinData.TextureID) == "Instance"
                and skinData.TextureID.TextureID
                or  skinData.TextureID
                or  ""
        end
    end

    local particle = SC_particles[skinName]
    if particle and handle then
        local p  = particle:Clone()
        p.Name   = "\0"
        p.Parent = handle
    end

    SC_activeSkins[gunName] = skinName
end

function SC_Reset(gunName)
    local tool = SC_GetTool(gunName)
    if not tool then return end

    local handle  = tool:FindFirstChild("Handle")
    local default = tool:FindFirstChild("Default")
    if not default then return end

    SC_Clean(default, handle)

    local orig = SC_originalData[gunName]
    if orig then
        default.Transparency = orig.Transparency
        default.TextureID    = orig.TextureID
        default.Massless     = orig.Massless
        default.CanCollide   = orig.CanCollide
        SC_originalData[gunName] = nil
    else
        default.Transparency = 0
        default.Massless     = false
        default.CanCollide   = true
    end

    if handle then handle:SetAttribute("SkinName", "Default") end
    SC_activeSkins[gunName] = nil
end

local function getHealthColor(pct)
    if pct > 0.6 then
        return Color3.fromRGB(0, 220, 80)
    elseif pct > 0.3 then
        return Color3.fromRGB(255, 180, 0)
    else
        return Color3.fromRGB(220, 40, 40)

local function esp(p, cr)
    local h   = cr:WaitForChild("Humanoid", 5)
    local hrp = cr:WaitForChild("HumanoidRootPart", 5)
    if not h or not hrp then return end

    local nameText = Drawing.new("Text")
    nameText.Visible = false
    nameText.Center  = true
    nameText.Outline = true
    nameText.Font    = 3
    nameText.Size    = 14
    nameText.Color   = espSettings.nameColor
    nameText.ZIndex  = 3

    local boxLines = {}
    for i = 1, 4 do
        local line = Drawing.new("Line")
        line.Visible   = false
        line.Color     = espSettings.boxColor
        line.Thickness = 1
        line.ZIndex    = 2
        table.insert(boxLines, line)
    end

    local filledBox = Drawing.new("Square")
    filledBox.Visible      = false
    filledBox.Color        = espSettings.filledColor
    filledBox.Filled       = true
    filledBox.Transparency = espSettings.filledTransparency
    filledBox.ZIndex       = 0

    local hBarBg = Drawing.new("Square")
    hBarBg.Visible      = false
    hBarBg.Color        = Color3.fromRGB(0, 0, 0)
    hBarBg.Filled       = true
    hBarBg.Transparency = 0.4
    hBarBg.ZIndex       = 1

    local hBarFill = Drawing.new("Square")
    hBarFill.Visible      = false
    hBarFill.Filled       = true
    hBarFill.Transparency = 0
    hBarFill.ZIndex       = 2

    local hpText = Drawing.new("Text")
    hpText.Visible = false
    hpText.Center  = true
    hpText.Outline = true
    hpText.Font    = 3
    hpText.Size    = 12
    hpText.Color   = Color3.fromRGB(255, 255, 255)
    hpText.ZIndex  = 3

    local conn, conn2, conn3

    local function dc()
        for _, d in ipairs({ nameText, filledBox, hBarBg, hBarFill, hpText }) do
            if d then d.Visible = false pcall(function() d:Remove() end) end
        end
        for _, line in ipairs(boxLines) do
            if line then line.Visible = false pcall(function() line:Remove() end) end
        end
        for _, cx in ipairs({ conn, conn2, conn3 }) do
            if cx then cx:Disconnect() end
        end
        conn, conn2, conn3 = nil, nil, nil
    end

    conn2 = cr.AncestryChanged:Connect(function(_, parent)
        if not parent then dc() end
    end)

    conn3 = h.HealthChanged:Connect(function(v)
        if v <= 0 or h:GetState() == Enum.HumanoidStateType.Dead then dc() end
    end)

    conn = rs.RenderStepped:Connect(function()
        pcall(function()
            local anyEnabled = espSettings.name or espSettings.box or espSettings.boxFilled
                            or espSettings.healthText

            local function hideAll()
                nameText.Visible  = false
                filledBox.Visible = false
                hBarBg.Visible    = false
                hBarFill.Visible  = false
                hpText.Visible    = false
                for _, l in ipairs(boxLines) do l.Visible = false end

            local headPos3D = rootPos + Vector3.new(0, 2.5, 0)

            local headScreen = c:WorldToViewportPoint(headPos3D)
            local feetScreen = c:WorldToViewportPoint(feetPos3D)

            if headScreen.Z <= 0 and feetScreen.Z <= 0 then
                hideAll() return
            end

            local boxHeight = math.abs(feetScreen.Y - headScreen.Y)
            local boxWidth  = boxHeight * 0.55
            local centerX   = (headScreen.X + feetScreen.X) / 2
            local centerY   = (headScreen.Y + feetScreen.Y) / 2
            local minY      = math.min(headScreen.Y, feetScreen.Y)

            local minX      = centerX - boxWidth / 2
            local maxX      = centerX + boxWidth / 2

            if boxWidth < 1 then
                hideAll() return
            end

            local maxHp = h.MaxHealth
            local curHp = h.Health
            local hpPct = maxHp > 0 and math.clamp(curHp / maxHp, 0, 1) or 0
            local hpCol = getHealthColor(hpPct)

            if espSettings.name then
                nameText.Position = Vector2.new(centerX, minY - 16)
                nameText.Text     = p.DisplayName
                nameText.Visible  = true
            else
                nameText.Visible = false
            end

            if espSettings.box then
                boxLines[1].From = Vector2.new(minX, minY)
                boxLines[1].To   = Vector2.new(maxX, minY)
                boxLines[2].From = Vector2.new(maxX, minY)
                boxLines[2].To   = Vector2.new(maxX, maxY)

                boxLines[3].To   = Vector2.new(minX, maxY)

                boxLines[4].To   = Vector2.new(minX, minY)

            else
                for _, l in ipairs(boxLines) do l.Visible = false end
            end

            if espSettings.boxFilled then

                filledBox.Position = Vector2.new(minX, minY)

function UpdateAuraColors(color)
    currentColor = color

    for i = 1, #particles do
        local part = particles[i]
        if part and part.Parent then
            local className = part.ClassName
            if className == "PointLight" then
                part.Color = color
            elseif className == "ParticleEmitter" or className == "Beam" or className == "Trail" then
                part.Color = colorSequence
            end

            for _, descendant in ipairs(part:GetDescendants()) do
                local dc = descendant.ClassName
                if dc == "PointLight" then
                    descendant.Color = color
                elseif dc == "ParticleEmitter" or dc == "Beam" or dc == "Trail" then
                    descendant.Color = colorSequence
                end
            end
        end
    end

            particles[i]:Destroy()

local FORCEFIELD_TRANSPARENCY = 0

local savedState = {}
local isEnabled = false
local currentColor = Color3.fromRGB(255, 255, 255)
local currentCharacter = nil

local function isPartOfTool(part)
    local current = part
    while current and current ~= currentCharacter do
        if current:IsA("Tool") then
            return true
        end
        current = current.Parent
    end
    return false
end

local function saveCharacterState(character)
    local state = {}
    for _, part in ipairs(character:GetDescendants()) do
        if part.Name == "HumanoidRootPart" then
            continue
        end
        if isPartOfTool(part) then
            continue
        end
        if part:IsA("BasePart") then
            state[part] = {
                Material = part.Material,
                Color = part.Color,
                Transparency = part.Transparency,
                TextureID = (part:IsA("MeshPart") and part.TextureID) or nil
            }
        elseif part:IsA("Shirt") or part:IsA("Pants") or part:IsA("ShirtGraphic") then
            if part:IsA("Shirt") then
                state[part] = { Template = part.ShirtTemplate }
            elseif part:IsA("Pants") then
                state[part] = { Template = part.PantsTemplate }
            elseif part:IsA("ShirtGraphic") then
                state[part] = { Graphic = part.Graphic }
            end
        end
    end
    return state
end

function applyForceField(character, state)
    for part, original in pairs(state) do
        if part:IsA("BasePart") then
            part.Material = Enum.Material.ForceField
            part.Color = currentColor
            part.Transparency = FORCEFIELD_TRANSPARENCY
            if part:IsA("MeshPart") then
                part.TextureID = ""
            end
        elseif part:IsA("Shirt") then
            part.ShirtTemplate = ""
        elseif part:IsA("Pants") then
            part.PantsTemplate = ""
        elseif part:IsA("ShirtGraphic") then
            part.Graphic = ""
        end
    end
end

function restoreCharacter(character, state)
    for part, original in pairs(state) do
        if part and part.Parent then
            if original.Material then
                part.Material = original.Material
                part.Color = original.Color
                part.Transparency = original.Transparency
                if part:IsA("MeshPart") and original.TextureID then
                    part.TextureID = original.TextureID
                end
            elseif original.Template then
                if part:IsA("Shirt") then
                    part.ShirtTemplate = original.Template
                elseif part:IsA("Pants") then
                    part.PantsTemplate = original.Template
                end
            elseif original.Graphic then
                part.Graphic = original.Graphic
            end
        end
    end
    savedState[character] = nil
end

function toggleEffect(enable)
    isEnabled = enable
    if currentCharacter then
        if isEnabled then
            if not savedState[currentCharacter] then
                savedState[currentCharacter] = saveCharacterState(currentCharacter)
            end
            applyForceField(currentCharacter, savedState[currentCharacter])
        else
            if savedState[currentCharacter] then
                restoreCharacter(currentCharacter, savedState[currentCharacter])
            end
        end
    end
end

local function updateChamsColor(color)
    currentColor = color
    if isEnabled and currentCharacter and savedState[currentCharacter] then
        for part, _ in pairs(savedState[currentCharacter]) do

local function RefreshConfigList()
    local configs = Menu:GetConfigs() or {}
    if ConfigList and ConfigList.Update then
        ConfigList.Value = ""
        ConfigList:Update(nil, configs)
    else
        warn("ConfigList object not found or Update method missing!")
        return
    end
    if #configs > 0 then
        SelectedConfig = configs[1]
        ConfigList:SetValue(SelectedConfig, configs)
    else
        SelectedConfig = nil
    end
end
local function FindConfigByPartialName(searchText)
    if not searchText or searchText == "" then
        return nil
    end
    local configs = Menu:GetConfigs() or {}
    local searchLower = searchText:lower()
    for _, configName in ipairs(configs) do
        if configName:lower() == searchLower then
            return configName
        end
    end
    for _, configName in ipairs(configs) do
        if configName:lower():find(searchLower, 1, true) then
            return configName
        end
    end
    
    return nil
end

            startPos = toggleBtn.Position

            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

        if input.UserInputType == Enum.UserInputType.Touch
            or input.UserInputType == Enum.UserInputType.MouseMovement then
            dragInput = input
