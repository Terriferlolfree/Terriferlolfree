local HttpService = game:GetService("HttpService")

-- KeyAuth API Functions
local function initSession()
	local url = "https://keyauth.win/api/1.2/?type=init&name=" .. KeyAuthApp.Name:gsub(" ", "%%20"):gsub("'", "%%27") .. "&ownerid=" .. KeyAuthApp.OwnerID .. "&ver=" .. KeyAuthApp.Version .. "&hash=" .. KeyAuthApp.Secret
	local success, response = pcall(function()
		return game:HttpGet(url)
	end)
	if not success then
		warn("Failed to connect to KeyAuth: " .. tostring(response))
		return nil
	end
	local data = HttpService:JSONDecode(response)
	if not data.success then
		warn("KeyAuth init failed: " .. tostring(data.message))
		return nil
	end
	return data.sessionid
end

local function verifyLicense(sessionid, key)
	if not sessionid then return {success = false, message = "Session initialization failed"} end
	local hwid = game:GetService("RbxAnalyticsService"):GetClientId()
	local url = "https://keyauth.win/api/1.2/?type=license&name=" .. KeyAuthApp.Name:gsub(" ", "%%20"):gsub("'", "%%27") .. "&ownerid=" .. KeyAuthApp.OwnerID .. "&sessionid=" .. sessionid .. "&key=" .. HttpService:UrlEncode(key) .. "&hwid=" .. hwid
	local success, response = pcall(function()
		return game:HttpGet(url)
	end)
	if not success then
		return {success = false, message = "Connection error"}
	end
	local parseSuccess, data = pcall(function()
		return HttpService:JSONDecode(response)
	end)
	if not parseSuccess then
		return {success = false, message = "Invalid response from server"}
	end
	return data
end

-- Initialize session
local sessionid = nil
local authenticated = false

local success, result = pcall(function()
	return initSession()
end)

if success then
	sessionid = result
end

if not sessionid then
	warn("KeyAuth initialization failed. Loading without key system...")
	authenticated = true -- Skip auth if KeyAuth fails
	-- Continue without key system if init fails
else
	-- Try to load saved key
	local savedKey = nil
	if readfile then
		local success, key = pcall(readfile, "staticcc_key.txt")
		if success and key and key ~= "" then
			savedKey = key
		end
	end
	
	-- If we have a saved key, try to verify it automatically
	if savedKey then
		local success, result = pcall(function()
			return verifyLicense(sessionid, savedKey)
		end)
		
		if success and result.success then
			-- Auto-login successful, skip GUI
			print("Auto-login successful!")
			-- Don't create GUI, just continue to main script
		else
			-- Saved key is invalid, delete it and show GUI
			if deletefile then pcall(deletefile, "staticcc_key.txt") end
			savedKey = nil
		end
	end
	
	-- Only show GUI if auto-login failed or no saved key
	if not savedKey then
	-- Destroy any existing KeyAuth GUIs
	pcall(function()
		if game:GetService("CoreGui"):FindFirstChild("KeyAuthGUI") then
			game:GetService("CoreGui"):FindFirstChild("KeyAuthGUI"):Destroy()
		end
	end)
	pcall(function()
		local pg = game:GetService("Players").LocalPlayer:FindFirstChild("PlayerGui")
		if pg and pg:FindFirstChild("KeyAuthGUI") then
			pg:FindFirstChild("KeyAuthGUI"):Destroy()
		end
	end)
	
	wait(0.5) -- Wait for old GUI to be destroyed

		end

	n.Size = UDim2.new(1, 0, 0, 34)

	Instance.new("UICorner", n).CornerRadius = UDim.new(0,4)

	local stroke = Instance.new("UIStroke", n)
	stroke.Color = T.border
	stroke.Thickness = 1

	local accent = Instance.new("Frame", n)
	accent.Size = UDim2.new(0, 3, 1, 0)
	accent.BackgroundColor3 = T.accent
	accent.BorderSizePixel = 0

	local lbl = Instance.new("TextLabel", n)
	lbl.Size = UDim2.new(1, -12, 1, 0)
	lbl.Position = UDim2.new(0, 10, 0, 0)
	lbl.BackgroundTransparency = 1
	lbl.Text = tostring(text)
	lbl.TextColor3 = T.text
	lbl.Font = Enum.Font.Code
	lbl.TextSize = 12
	lbl.TextXAlignment = Enum.TextXAlignment.Left

	-- start offscreen
	n.Position = UDim2.new(1, 40, 0, 0)

	TweenService:Create(
		n,
		TweenInfo.new(0.22, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
		{
			Position = UDim2.new(0,0,0,0)
		}
	):Play()

	task.delay(duration, function()
		local outTween = TweenService:Create(
			n,
			TweenInfo.new(0.22, Enum.EasingStyle.Quad, Enum.EasingDirection.In),
			{
				Position = UDim2.new(1, 40, 0, 0),
				BackgroundTransparency = 1
			}
		)

		outTween:Play()

		outTween.Completed:Wait()

		n:Destroy()
	end)
end

-- Helper function to check if input matches configured key (supports KeyCode, UserInputType, and custom mouse buttons)
local function inputMatches(input, configKey)
	if typeof(configKey)=="EnumItem" then
		if tostring(configKey):find("KeyCode") then
			return input.KeyCode == configKey
		elseif tostring(configKey):find("UserInputType") then
			return input.UserInputType == configKey
		end
	end
	return false
end

		if typeof(input)=="EnumItem" then
			if tostring(input):find("KeyCode") then
				return tostring(input):gsub("Enum.KeyCode.","")
			elseif tostring(input):find("UserInputType") then
				local mouseMap={
					[Enum.UserInputType.MouseButton1]="Mouse1",
					[Enum.UserInputType.MouseButton2]="Mouse2",
					[Enum.UserInputType.MouseButton3]="Mouse3"
				}
				return mouseMap[input] or tostring(input):gsub("Enum.UserInputType.","")
			end
		end
		return tostring(input)
	end
	
	keyLbl.Text="["..getDisplayText(cfg[key]).."]"
	Instance.new("UICorner",keyLbl).CornerRadius=UDim.new(0,2)
	keyLbl.MouseButton1Click:Connect(function()
		if listening then return end;listening=true;keyLbl.Text="[...]";keyLbl.TextColor3=Color3.fromRGB(255,200,60)
		local conn;conn=UIS.InputBegan:Connect(function(i,gp)
			if gp then return end
			-- Accept keyboard keys
			if i.UserInputType==Enum.UserInputType.Keyboard then
				cfg[key]=i.KeyCode;keyLbl.Text="["..getDisplayText(i.KeyCode).."]"
				keyLbl.TextColor3=T.text;listening=false;conn:Disconnect()

local function isAlive(p) 
	local h=getHum(p)
	if not h or h.Health<=0 then return false end
	-- Da Hood/Hood Customs KO check - check for BodyEffects folder
	local c=p.Character
	if c then
		local bodyEffects=c:FindFirstChild("BodyEffects")
		if bodyEffects then
			local ko=bodyEffects:FindFirstChild("K.O")
			if ko and ko.Value==true then return false end
		end
	end

local function sameTeam(p) return lp.Team and p.Team and lp.Team==p.Team end

local function wallBetween(pos)
	local rp=RaycastParams.new();rp.FilterType=Enum.RaycastFilterType.Exclude
	local chars={};for _,pl in pairs(Players:GetPlayers()) do if pl.Character then table.insert(chars,pl.Character) end end
	rp.FilterDescendantsInstances=chars
	return workspace:Raycast(cam.CFrame.Position,(pos-cam.CFrame.Position),rp)~=nil
end
local function getPredicted(part)
	if not part then return nil end
	local v=part.AssemblyLinearVelocity
	local predY = cfg.camlockPredY
	
	-- Check if target is jumping or falling
	if camlockTarget then
		local char = camlockTarget.Character
		if char then
			local hum = char:FindFirstChildOfClass("Humanoid")
			if hum then
				-- Check if jumping (moving upward and not on ground)
				if cfg.camlockJumpPredEnabled and v.Y > 5 and hum.FloorMaterial == Enum.Material.Air then
					predY = cfg.camlockJumpPredY
				-- Check if falling (moving downward and not on ground)
				elseif cfg.camlockFallPredEnabled and v.Y < -5 and hum.FloorMaterial == Enum.Material.Air then
					predY = cfg.camlockFallPredY
				end
			end
		end
	end
	
	return part.Position+Vector3.new(v.X*cfg.camlockPredX,v.Y*predY,v.Z*cfg.camlockPredZ)
end
local function closestToCursor(fovR,teamCheck,maxDist)
	local best,bestD=nil,fovR or math.huge
	local center=Vector2.new(mouse.X,mouse.Y)
	local distLimit=maxDist or cfg.camlockMaxDist
	for _,p in pairs(Players:GetPlayers()) do
		if p==lp then continue end
		if teamCheck and sameTeam(p) then continue end
		if not isAlive(p) then continue end
		local hrp=getHRP(p);if not hrp then continue end
		if (hrp.Position-cam.CFrame.Position).Magnitude>distLimit then continue end
		local sp,on=cam:WorldToViewportPoint(hrp.Position);if not on then continue end
		local d=(Vector2.new(sp.X,sp.Y)-center).Magnitude
		if d<bestD then bestD=d;best=p end
	end
	return best
end

local BONES={{"Head","UpperTorso"},{"UpperTorso","LowerTorso"},{"UpperTorso","RightUpperArm"},{"RightUpperArm","RightLowerArm"},{"RightLowerArm","RightHand"},{"UpperTorso","LeftUpperArm"},{"LeftUpperArm","LeftLowerArm"},{"LeftLowerArm","LeftHand"},{"LowerTorso","RightUpperLeg"},{"RightUpperLeg","RightLowerLeg"},{"RightLowerLeg","RightFoot"},{"LowerTorso","LeftUpperLeg"},{"LeftUpperLeg","LeftLowerLeg"},{"LeftLowerLeg","LeftFoot"}}
local function newDraw(t,props) local d=Drawing.new(t);for k,v in pairs(props) do d[k]=v end;return d end
local function createESP(plr)
	if plr==lp or espData[plr] then return end
	local d={}
	d.hl=Instance.new("Highlight");d.hl.FillTransparency=1;d.hl.OutlineColor=T.accent;d.hl.Enabled=false
	d.box=newDraw("Square",{Thickness=1.5,Color=T.accent,Filled=false,Visible=false})
	d.boxFill=newDraw("Square",{Thickness=1,Color=T.accent,Filled=true,Transparency=cfg.espFillTransparency,Visible=false})
	d.name=newDraw("Text",{Size=12,Color=Color3.fromRGB(255,255,255),Outline=true,Center=true,Visible=false})
	d.dist=newDraw("Text",{Size=10,Color=T.dim,Outline=true,Center=true,Visible=false})
	d.hbg=newDraw("Square",{Thickness=1,Color=Color3.fromRGB(0,0,0),Filled=true,Visible=false})
	d.hfill=newDraw("Square",{Thickness=1,Color=Color3.fromRGB(100,255,150),Filled=true,Visible=false})
	d.headDot=newDraw("Circle",{Thickness=1,NumSides=16,Radius=4,Color=T.accent,Filled=true,Visible=false})
	d.tracer=newDraw("Line",{Thickness=1,Color=T.accent,Visible=false})
	d.bones={};for _=1,#BONES do table.insert(d.bones,newDraw("Line",{Thickness=1,Color=T.accent,Visible=false})) end
	espData[plr]=d
	plr.CharacterAdded:Connect(function() 
		if cfg.esp and cfg.espChams then 
			d.hl.Adornee=plr.Character
			d.hl.Parent=plr.Character
			d.hl.Enabled=true
		end 
	end)
end
local function removeESP(plr)
	local d=espData[plr];if not d then return end
	d.hl:Destroy();d.box:Remove();d.boxFill:Remove();d.name:Remove();d.dist:Remove()
	d.hbg:Remove();d.hfill:Remove();d.headDot:Remove();d.tracer:Remove()
	for _,b in pairs(d.bones) do b:Remove() end;espData[plr]=nil
end
local function toggleESP(state)
	if state then for _,p in pairs(Players:GetPlayers()) do createESP(p) end
	else for p in pairs(espData) do removeESP(p) end end
end
Players.PlayerAdded:Connect(function(p) if cfg.esp then createESP(p) end end)
Players.PlayerRemoving:Connect(removeESP)

-- Weapon Chams
local activeWeaponHighlights={}

local function getWeaponChamsColor()
	if cfg.weaponChamsRainbow then
		return Color3.fromHSV((tick()%5)/5,1,1)
	end
	return cfg.weaponChamsColor
end

local function weaponAllowed(name)
	name=string.lower(name)
	if name:find("rev") then return true end
	if name:find("db") then return true end
	if name:find("tac") then return true end
	if name:find("revolver") then return true end
	if name:find("double") then return true end
	if name:find("shotgun") then return true end
	return false
end

local function removeWeaponGlow(model)
	local glow=activeWeaponHighlights[model]
	if glow then
		glow:Destroy()
		activeWeaponHighlights[model]=nil
	end
end

local function addWeaponGlow(model)
	if activeWeaponHighlights[model] then return end
	local glow=Instance.new("Highlight")
	glow.Name="WeaponGlow"
	glow.Adornee=model
	glow.DepthMode=Enum.HighlightDepthMode.AlwaysOnTop
	glow.OutlineTransparency=0
	glow.FillTransparency=cfg.weaponChamsFill and 0.65 or 1
	glow.OutlineColor=getWeaponChamsColor()
	glow.FillColor=getWeaponChamsColor()
	glow.Parent=model
	activeWeaponHighlights[model]=glow
end

local function refreshWeaponGlow()
	for model,glow in pairs(activeWeaponHighlights) do
		if not model or not model.Parent then
			activeWeaponHighlights[model]=nil
		else
			glow.OutlineColor=getWeaponChamsColor()
			glow.FillColor=getWeaponChamsColor()
			glow.FillTransparency=cfg.weaponChamsFill and 0.65 or 1
		end
	end
end

local function scanCharacterWeapons()
	local character=lp.Character
	if not character then return end
	for _,item in ipairs(character:GetChildren()) do
		if item:IsA("Tool") and weaponAllowed(item.Name) then
			addWeaponGlow(item)
		end
	end
end

local function scanCameraWeapons()
	for _,item in ipairs(cam:GetChildren()) do
		if item:IsA("Model") and weaponAllowed(item.Name) then
			addWeaponGlow(item)
		end
	end
end

local function updateWeaponChams()
	if not cfg.weaponChams then
		for model in pairs(activeWeaponHighlights) do
			removeWeaponGlow(model)
		end
		return
	end
	scanCharacterWeapons()
	scanCameraWeapons()
end

-- Rain (Drawing-based for better performance)
local rainDrops={}
local rainbowHue=0

local function getRainbowColor()
	rainbowHue=(rainbowHue+0.01)%1
	return Color3.fromHSV(rainbowHue,1,1)
end

local function rebuildRain()
	-- Clear old drops
	for _,d in pairs(rainDrops) do
		if d.line then d.line:Remove() end
	end
	rainDrops={}
	
	-- Create new drops
	for i=1,cfg.rainAmount do
		local line=Drawing.new("Line")
		line.Thickness=1
		line.Transparency=1-cfg.rainTransparency
		line.Color=cfg.rainColor
		line.Visible=cfg.rainEnabled
		
		local offset=Vector3.new(
			math.random(-cfg.rainRadius,cfg.rainRadius),
			math.random(0,35),
			math.random(-cfg.rainRadius,cfg.rainRadius)
		)
		
		table.insert(rainDrops,{
			line=line,
			offset=offset,
			yPos=math.random(0,50)
		})
	end
end

rebuildRain()

RunService.RenderStepped:Connect(function(dt)
	if not cfg.rainEnabled then return end
	
	local char=lp.Character
	if not char then return end
	local hrp=char:FindFirstChild("HumanoidRootPart")
	if not hrp then return end
	
	for _,drop in pairs(rainDrops) do
		if drop.line then
			-- Update Y position
			drop.yPos=drop.yPos-(cfg.rainSpeed*dt*60)
			
			-- Reset if below ground
			if drop.yPos<-10 then
				drop.yPos=50
				drop.offset=Vector3.new(
					math.random(-cfg.rainRadius,cfg.rainRadius),
					0,
					math.random(-cfg.rainRadius,cfg.rainRadius)
				)
			end
			
			-- Calculate positions
			local basePos=hrp.Position+drop.offset
			local startPos=basePos+Vector3.new(0,drop.yPos,0)
			local endPos=startPos-Vector3.new(0,cfg.rainLength,0)
			
			-- Convert to screen space
			local startScreen,startOnScreen=cam:WorldToViewportPoint(startPos)
			local endScreen,endOnScreen=cam:WorldToViewportPoint(endPos)
			
			if startOnScreen or endOnScreen then
				drop.line.From=Vector2.new(startScreen.X,startScreen.Y)
				drop.line.To=Vector2.new(endScreen.X,endScreen.Y)
				drop.line.Visible=true
				
				-- Rainbow mode
				if cfg.rainbowRain then
					drop.line.Color=getRainbowColor()
				else
					drop.line.Color=cfg.rainColor
				end
				
				drop.line.Transparency=1-cfg.rainTransparency
			else
				drop.line.Visible=false
			end
		end
	end
end)

-- Flight / Noclip / Spin
local function startFlight()
	local char=lp.Character;if not char then return end
	local hrp=char:FindFirstChild("HumanoidRootPart");if not hrp then return end
	local bg=Instance.new("BodyGyro",hrp);bg.MaxTorque=Vector3.new(1e5,1e5,1e5);bg.P=1e4;bg.Name="FlyGyro"
	local bv=Instance.new("BodyVelocity",hrp);bv.MaxForce=Vector3.new(1e5,1e5,1e5);bv.Velocity=Vector3.zero;bv.Name="FlyVel"
	flightBody={bg=bg,bv=bv}
end
local function stopFlight()
	local char=lp.Character
	if char then
		local hrp=char:FindFirstChild("HumanoidRootPart")
		if hrp then
			local bg=hrp:FindFirstChild("FlyGyro");local bv=hrp:FindFirstChild("FlyVel")
			if bg then bg:Destroy() end;if bv then bv:Destroy() end
		end
	end
	flightBody=nil
end
local function enableNoclip() noclipConn=RunService.Stepped:Connect(function() local char=lp.Character;if not char then return end;for _,p in pairs(char:GetDescendants()) do if p:IsA("BasePart") then p.CanCollide=false end end end) end
local function disableNoclip() if noclipConn then noclipConn:Disconnect();noclipConn=nil end end

-- Matrix Hub style walkspeed
local function startWalkspeed()
	if walkspeedConn then return end
	walkspeedConn=RunService.Heartbeat:Connect(function()
		if not walkspeedEnabled then return end
		local char=lp.Character
		if not char then return end
		local hum=char:FindFirstChildOfClass("Humanoid")
		local hrp=char:FindFirstChild("HumanoidRootPart")
		if not hum or not hrp then return end
		
		local moveVector=hum.MoveDirection
		if moveVector.Magnitude>0 then
			hrp.Velocity=Vector3.new(moveVector.X*cfg.walkspeed,hrp.Velocity.Y,moveVector.Z*cfg.walkspeed)
		end
	end)
end
local function stopWalkspeed()
	if walkspeedConn then walkspeedConn:Disconnect();walkspeedConn=nil end
	walkspeedEnabled=false
end

-- Infinite Jump (no cooldown)
local infiniteJumpConn=nil
local function startInfiniteJump()
	if infiniteJumpConn then infiniteJumpConn:Disconnect() end
	infiniteJumpConn=RunService.RenderStepped:Connect(function()
		if not cfg.infiniteJump then return end
		local char=lp.Character
		if not char then return end
		local hum=char:FindFirstChild("Humanoid")
		if not hum then return end
		hum.UseJumpPower=true
		hum.JumpPower=50
		hum.JumpHeight=7.2
		hum:SetStateEnabled(Enum.HumanoidStateType.Jumping,true)
	end)
end
local function stopInfiniteJump()
	if infiniteJumpConn then infiniteJumpConn:Disconnect();infiniteJumpConn=nil end
end

-- Anti-cheat bypass jump power using velocity manipulation
local function setupJumpPower()
	if jumpConnection then jumpConnection:Disconnect() end
	jumpConnection=UIS.InputBegan:Connect(function(input,gameProcessed)
		if gameProcessed then return end
		if input.KeyCode==Enum.KeyCode.Space then
			-- Only use velocity manipulation if jump power is above default (50)
			if cfg.jumppower<=50 then return end
			
			local char=lp.Character
			if not char then return end
			local hum=char:FindFirstChildOfClass("Humanoid")
			local hrp=char:FindFirstChild("HumanoidRootPart")
			if not hum or not hrp then return end
			
			-- Check if player is on ground (can jump)
			if hum.FloorMaterial~=Enum.Material.Air then
				-- Calculate jump velocity based on configured jump power
				-- Standard Roblox jump formula: velocity = sqrt(2 * jumpPower * workspace.Gravity)
				local jumpVelocity=math.sqrt(2*cfg.jumppower*workspace.Gravity)
				hrp.Velocity=Vector3.new(hrp.Velocity.X,jumpVelocity,hrp.Velocity.Z)
			end
		end
	end)
end

-- Always-on low friction when walkspeed is active (feels like external slippery speed)
local function applySlippery()
	local char=lp.Character;if not char then return end
	for _,p in pairs(char:GetDescendants()) do
		if p:IsA("BasePart") then
			p.CustomPhysicalProperties=PhysicalProperties.new(0.7,0.01,0.1,0,0)
		end
	end
end
local function removeSlippery()
	local char=lp.Character;if not char then return end
	for _,p in pairs(char:GetDescendants()) do
		if p:IsA("BasePart") then
			p.CustomPhysicalProperties=PhysicalProperties.new(0.7,0.3,0.5,0,0)
		end
	end
end
local function do360()
	if spinActive then return end;spinActive=true
	local steps=60*cfg.spin360Loops
	local anglePerStep=(360*cfg.spin360Direction)/60
	local waitTime=(1/cfg.spin360Speed)/60
	for i=1,steps do if not spinActive then break end;cam.CFrame=cam.CFrame*CFrame.Angles(0,math.rad(anglePerStep),0);task.wait(waitTime) end
	spinActive=false
end

local inputConnection = UIS.InputBegan:Connect(function(i,gp)
	if gp then return end
	if i.KeyCode==cfg.camlockKey then
		if not cfg.camlock then return end
		if cfg.camlockMode=="Hold" then
			local fovR=cfg.camlockFOVEnabled and cfg.camlockFOVRadius or nil
			camlockTarget=closestToCursor(fovR,cfg.camlockTeamCheck,cfg.camlockMaxDist)
		elseif cfg.camlockMode=="Toggle" then
			if camlockTarget then
				camlockTarget=nil
				cam.CameraType=Enum.CameraType.Custom
			else
				local fovR=cfg.camlockFOVEnabled and cfg.camlockFOVRadius or nil
				camlockTarget=closestToCursor(fovR,cfg.camlockTeamCheck,cfg.camlockMaxDist)
			end
		end
	end
	-- Silent Aim Key Handling
	if cfg.silentAimKey~=Enum.KeyCode.Unknown and i.KeyCode==cfg.silentAimKey then
		if cfg.silentAimMode=="Toggle" then
			cfg.silentAim=not cfg.silentAim
		elseif cfg.silentAimMode=="Hold" then
			cfg.silentAim=true
		elseif cfg.silentAimMode=="Always" then
			-- Always mode: lock to closest on key press
			silentTarget=closestToCursor(cfg.silentAimFOV,cfg.silentAimTeamCheck,cfg.silentAimMaxDist)
		end
	end
	if i.KeyCode==cfg.spin360Key then task.spawn(do360) end
	if cfg.flightKey~=Enum.KeyCode.Unknown and i.KeyCode==cfg.flightKey then
		cfg.flightEnabled=not cfg.flightEnabled
		if cfg.flightEnabled then startFlight() else stopFlight() end
	end
	if cfg.walkspeedKey~=Enum.KeyCode.Unknown and i.KeyCode==cfg.walkspeedKey then
		walkspeedEnabled=not walkspeedEnabled
		if walkspeedEnabled then
			startWalkspeed()
		else
			stopWalkspeed()
		end
	end
	if cfg.jumppowerKey~=Enum.KeyCode.Unknown and i.KeyCode==cfg.jumppowerKey then
		-- Toggle between custom jump power and default
		if cfg.jumppower~=50 then
			cfg.jumppower=50
		else
			cfg.jumppower=100 -- or whatever default enhanced value you want
		end
	end
end)

UIS.InputEnded:Connect(function(i)
	-- Silent Aim Hold Mode Release
	if cfg.silentAimKey~=Enum.KeyCode.Unknown and i.KeyCode==cfg.silentAimKey then
		if cfg.silentAimMode=="Hold" then
			cfg.silentAim=false
		end
	end
end)

local origL={Brightness=Lighting.Brightness,ClockTime=Lighting.ClockTime,GlobalShadows=Lighting.GlobalShadows,FogStart=Lighting.FogStart,FogEnd=Lighting.FogEnd,FogColor=Lighting.FogColor}
local fullbright=false
local fullbrightConn=nil
local lightingConn=nil
local function toggleFB() 
	fullbright=not fullbright
	if fullbright then 
		Lighting.Brightness=5
		Lighting.ClockTime=12
		if fullbrightConn then fullbrightConn:Disconnect() end
		fullbrightConn=RunService.RenderStepped:Connect(function()
			Lighting.Brightness=5
			Lighting.ClockTime=12
		end)
	else 
		if fullbrightConn then fullbrightConn:Disconnect();fullbrightConn=nil end
		Lighting.Brightness=origL.Brightness
		Lighting.ClockTime=origL.ClockTime
	end 
end
local function setLightingMode(mode)
	if lightingConn then lightingConn:Disconnect();lightingConn=nil end
	if mode=="Day" then
		Lighting.ClockTime=14
		Lighting.Brightness=2
		lightingConn=RunService.RenderStepped:Connect(function()
			Lighting.ClockTime=14
			Lighting.Brightness=2
		end)
	elseif mode=="Night" then
		Lighting.ClockTime=0
		Lighting.Brightness=1
		lightingConn=RunService.RenderStepped:Connect(function()
			Lighting.ClockTime=0
			Lighting.Brightness=1
		end)
	elseif mode=="Sunset" then
		Lighting.ClockTime=18
		lightingConn=RunService.RenderStepped:Connect(function()
			Lighting.ClockTime=18
		end)
	end
	cfg.lightingMode=mode
end
local function setFog(on) if on then Lighting.FogStart=cfg.fogStart;Lighting.FogEnd=cfg.fogEnd;Lighting.FogColor=cfg.fogColor else Lighting.FogStart=origL.FogStart;Lighting.FogEnd=origL.FogEnd;Lighting.FogColor=origL.FogColor end end

local origSky=nil
local rotatingSkyConn=nil
local function applySky(preset) 
	local sky=Lighting:FindFirstChildOfClass("Sky") or Instance.new("Sky",Lighting)
	if not origSky then 
		origSky={SkyboxBk=sky.SkyboxBk,SkyboxDn=sky.SkyboxDn,SkyboxFt=sky.SkyboxFt,SkyboxLf=sky.SkyboxLf,SkyboxRt=sky.SkyboxRt,SkyboxUp=sky.SkyboxUp} 
	end
	local src=preset==nil and origSky or skyboxes[preset]
	if src then 
		for k,v in pairs(src) do 
			sky[k]=v 
		end 
	end
	cfg.selectedSkybox=preset
end
local function toggleRotatingSkybox(enabled)
	if rotatingSkyConn then
		rotatingSkyConn:Disconnect()
		rotatingSkyConn=nil
	end
	if enabled then
		local sky=Lighting:FindFirstChildOfClass("Sky") or Instance.new("Sky",Lighting)
		rotatingSkyConn=RunService.PreRender:Connect(function(delta)
			sky.SkyboxOrientation=sky.SkyboxOrientation+Vector3.new(0,30*delta,0)
		end)
	else
		local sky=Lighting:FindFirstChildOfClass("Sky")
		if sky then sky.SkyboxOrientation=Vector3.new(0,0,0) end
	end
end

local origMats={}
local origColors={}

local function applyTextureColor(color)
	cfg.textureColor=color
	cfg.textureColorEnabled=true

	for _,obj in pairs(workspace:GetDescendants()) do
		if obj:IsA("BasePart") then
			if lp.Character and obj:IsDescendantOf(lp.Character) then continue end
			if not origColors[obj] then origColors[obj]=obj.Color end
			obj.Color=color
		elseif obj:IsA("Decal") or obj:IsA("Texture") then
			if not origColors[obj] then origColors[obj]=obj.Color3 end
			obj.Color3=color
		end
	end
end

local function resetTextureColors()
	for obj,color in pairs(origColors) do
		if obj and obj.Parent then
			if obj:IsA("BasePart") then
				obj.Color=color
			elseif obj:IsA("Decal") or obj:IsA("Texture") then
				obj.Color3=color
			end
		end
	end

	origColors={}
	cfg.textureColorEnabled=false
end

local function applyMat(mat)
	for _,p in pairs(workspace:GetDescendants()) do
		if p:IsA("BasePart") then
			if lp.Character and p:IsDescendantOf(lp.Character) then continue end
			if not origMats[p] then origMats[p]=p.Material end
			p.Material=mat
		end
	end
end

local function resetMats()
	for p,m in pairs(origMats) do
		if p and p.Parent then p.Material=m end
	end
	origMats={}
end

-- Universal Silent Aim (works on all executors including external ones)
local silentTarget=nil
local silentAimHooksInstalled=false
local oldIndex=nil
local oldNamecall=nil

local function installSilentAimHooks()
	if silentAimHooksInstalled then return end
	silentAimHooksInstalled=true
	
	oldIndex=hookmetamethod and hookmetamethod(game,"__index",newcclosure(function(self,key)
		-- Check if silent aim should be active based on mode
		local silentActive=cfg.silentAim or cfg.silentAimMode=="Always"
		if silentActive and checkcaller and not checkcaller() then
			-- Hit chance check
			if math.random(1,100)>cfg.silentAimHitChance then
				return oldIndex(self,key)
			end
			if key=="Hit" or key=="Target" then
				local target=closestToCursor(cfg.silentAimFOV,cfg.silentAimTeamCheck,cfg.silentAimMaxDist)
				if target then
					local char=target.Character
					if char then
						local hum=char:FindFirstChildOfClass("Humanoid")
						local part
						-- Closest part logic bypasses hit part selection
						if cfg.silentAimClosestPart then
							local closestDist=math.huge
							for _,p in pairs(char:GetChildren()) do
								if p:IsA("BasePart") then
									local d=(p.Position-cam.CFrame.Position).Magnitude
									if d<closestDist then
										closestDist=d
										part=p
									end
								end
							end
						else
							local partName=cfg.silentAimPart
							if cfg.silentAimUseAirPart and hum and hum.FloorMaterial==Enum.Material.Air then
								partName=cfg.silentAimAirPart
							end
							part=char:FindFirstChild(partName) or char:FindFirstChild("HumanoidRootPart")
						end
						-- KO check - skip if enabled and target is knocked
						if part and cfg.silentAimKOCheck then
							local bodyEffects=char:FindFirstChild("BodyEffects")
							if bodyEffects then
								local ko=bodyEffects:FindFirstChild("K.O")
								if ko and ko.Value==true then
									return oldIndex(self,key)
								end
							end
							if hum and hum.Health<=0 then
								return oldIndex(self,key)
							end
						end
						if part then
							local dist=(part.Position-cam.CFrame.Position).Magnitude
							if dist<=cfg.silentAimMaxDist then
								if not(cfg.silentAimWallCheck and wallBetween(part.Position)) then
									local v=part.AssemblyLinearVelocity
									local pred=part.Position+Vector3.new(v.X*cfg.silentAimPredX,v.Y*cfg.silentAimPredY,v.X*cfg.silentAimPredX)
									if key=="Hit" then
										return CFrame.new(pred)
									elseif key=="Target" then
										return part
									end
								end
							end
						end
					end
				end
			end
		end
		return oldIndex(self,key)
	end)) or nil
	
	-- Fallback for executors without hookmetamethod (external executors)
	if not oldIndex then
		local mt=getrawmetatable(game)
		if mt then
			oldNamecall=mt.__namecall
			setreadonly(mt,false)
			mt.__namecall=newcclosure(function(self,...)
				local method=getnamecallmethod()
				local args={...}
				local silentActive=cfg.silentAim or cfg.silentAimMode=="Always"
				if silentActive and (method=="FireServer" or method=="InvokeServer") then
					-- Hit chance check
					if math.random(1,100)<=cfg.silentAimHitChance then
						local target=closestToCursor(cfg.silentAimFOV,cfg.silentAimTeamCheck,cfg.silentAimMaxDist)
						if target then
							local char=target.Character
							if char then
								local hum=char:FindFirstChildOfClass("Humanoid")
								local part
								-- Closest part logic bypasses hit part selection
								if cfg.silentAimClosestPart then
									local closestDist=math.huge
									for _,p in pairs(char:GetChildren()) do
										if p:IsA("BasePart") then
											local d=(p.Position-cam.CFrame.Position).Magnitude
											if d<closestDist then
												closestDist=d
												part=p
											end
										end
									end
								else
									local partName=cfg.silentAimPart
									if cfg.silentAimUseAirPart and hum and hum.FloorMaterial==Enum.Material.Air then
										partName=cfg.silentAimAirPart
									end
									part=char:FindFirstChild(partName) or char:FindFirstChild("HumanoidRootPart")
								end
								-- KO check - skip if enabled and target is knocked
								local isKnockedOrDead=false
								if cfg.silentAimKOCheck then
									local bodyEffects=char:FindFirstChild("BodyEffects")
									if bodyEffects then
										local ko=bodyEffects:FindFirstChild("K.O")
										if ko and ko.Value==true then
											isKnockedOrDead=true
										end
									end
									if hum and hum.Health<=0 then
										isKnockedOrDead=true
									end
								end
								if part and not isKnockedOrDead then
									local dist=(part.Position-cam.CFrame.Position).Magnitude
									if dist<=cfg.silentAimMaxDist then
										if not(cfg.silentAimWallCheck and wallBetween(part.Position)) then
											local v=part.AssemblyLinearVelocity
											local pred=part.Position+Vector3.new(v.X*cfg.silentAimPredX,v.Y*cfg.silentAimPredY,v.X*cfg.silentAimPredX)
											-- Replace position arguments in remote calls
											for i,v in pairs(args) do
												if typeof(v)=="Vector3" then
													args[i]=pred
													break
												elseif typeof(v)=="CFrame" then
													args[i]=CFrame.new(pred)
													break
												end
											end
										end
									end
								end
							end
						end
					end
				end
				return oldNamecall(self,unpack(args))
			end)
			setreadonly(mt,true)
		end
	end
end

-- Config
local function saveConfig(name)
	if not writefile then warn("writefile not supported") return end
	local data={}
	for k,v in pairs(cfg) do
		local t=type(v)
		if t=="boolean" or t=="number" or t=="string" then
			data[k]=v
		elseif t=="userdata" then
			-- Color3
			local okC,r,g,b=pcall(function() return v.R,v.G,v.B end)
			if okC then data[k]={_type="Color3",r=r,g=g,b=b};continue end
			-- EnumItem (KeyCode, Material, etc)
			local okE,ev=pcall(function() return v.Value end)
			if okE then data[k]={_type="EnumItem",enum=tostring(v)};continue end
		elseif v==nil then
			-- Explicitly save nil values (like currentMaterial when not set)
			data[k]=nil
		end
	end
	local ok,err=pcall(writefile,"staticcc_"..name..".json",HttpService:JSONEncode(data))
	if not ok then warn("Save failed: "..tostring(err)) end
end
local function deleteConfig(name)
	if not name or name=="" then return end

	local deleteFn=deletefile or delfile
	if not deleteFn then
		warn("deletefile/delfile not supported")
		return
	end

	local ok,err=pcall(deleteFn,"staticcc_"..name..".json")
	if ok then
		notify("Deleted config: "..name,2)
	else
		warn("Delete failed: "..tostring(err))
	end
end

local function loadConfig(name)
	if not readfile then warn("readfile not supported") return end
	local ok,raw=pcall(readfile,"staticcc_"..name..".json")
	if not ok then warn("Config not found: "..name) return end
	local ok2,data=pcall(function() return HttpService:JSONDecode(raw) end)
	if not ok2 then warn("Config parse error") return end
	for k,v in pairs(data) do
		if type(v)=="table" then
			if v._type=="Color3" then cfg[k]=Color3.new(v.r,v.g,v.b)
			elseif v._type=="EnumItem" then
				-- parse "Enum.KeyCode.Q" -> Enum.KeyCode.Q
				local enumStr=tostring(v.enum)
				local ok3,ev=pcall(function()
					local parts=enumStr:split(".")
					if #parts==3 then return Enum[parts[2]][parts[3]] end
				end)
				if ok3 and ev then cfg[k]=ev end
			end
		else cfg[k]=v end
	end
	-- Apply states after loading
	if cfg.noclip then enableNoclip() else disableNoclip() end
	if cfg.fogEnabled then 
		task.wait(0.1) -- Small delay to ensure lighting is ready
		setFog(true) 
	else 
		setFog(false) 
	end
	if cfg.fullbrightEnabled then toggleFB() end
	if cfg.currentMaterial then applyMat(cfg.currentMaterial) end
	if cfg.selectedSkybox then applySky(cfg.selectedSkybox) end
	if cfg.rotatingSkybox then toggleRotatingSkybox(true) end
	if cfg.lightingMode then setLightingMode(cfg.lightingMode) end
	if cfg.fpsBoostEnabled then fpsBoost() end
	if cfg.rainEnabled then rebuildRain() end
	updateWeaponChams()
	-- Install silent aim hooks if enabled or in Always mode (deferred to ensure function exists)
	task.spawn(function()
		task.wait(0.2) -- Wait for script to fully load
		if cfg.silentAim or cfg.silentAimMode=="Always" then 
			if installSilentAimHooks then
				installSilentAimHooks() 
			end
		end
	end)
end
local function listConfigs()
	if not listfiles then return {} end
	local ok,files=pcall(listfiles,"")
	if not ok then return {} end
	local out={}
	for _,f in pairs(files) do
		local n=tostring(f):match("staticcc_(.+)%.json$")
		if n then table.insert(out,n) end
	end
	return out
end

	end

	local char = camlockTarget.Character
	if not char then camlockTarget=nil;cam.CameraType=Enum.CameraType.Custom;return end
	local hum = char:FindFirstChildOfClass("Humanoid")
	if not hum then camlockTarget=nil;cam.CameraType=Enum.CameraType.Custom;return end
	
	-- KO check - unlock if target is knocked or dead
	if cfg.camlockKOCheck then
		local bodyEffects=char:FindFirstChild("BodyEffects")
		if bodyEffects then
			local ko=bodyEffects:FindFirstChild("K.O")
			if ko and ko.Value==true then
				camlockTarget=nil
				cam.CameraType=Enum.CameraType.Custom
				return
			end
		end
		if hum.Health<=0 then
			camlockTarget=nil
			cam.CameraType=Enum.CameraType.Custom
			return
		end
	end

	local partName = cfg.camlockPart
	if cfg.camlockUseAirPart and hum.FloorMaterial==Enum.Material.Air then
		partName = cfg.camlockAirPart
	end
	local part
	-- Closest part logic bypasses hit part selection
	if cfg.camlockClosestPart then
		local closestDist=math.huge
		for _,p in pairs(char:GetChildren()) do
			if p:IsA("BasePart") then
				local d=(p.Position-cam.CFrame.Position).Magnitude
				if d<closestDist then
					closestDist=d
					part=p
				end
			end
		end
	else
		part = char:FindFirstChild(partName) or char:FindFirstChild("HumanoidRootPart")
	end
	if not part then return end
	if cfg.camlockWallCheck and wallBetween(part.Position) then return end
	local pred = getPredicted(part)
	if not pred then return end
	local from = cam.CFrame
	local to = CFrame.new(from.Position, pred)
	local t = cfg.camlockSmooth
	local style = cfg.camlockAimStyle
	if style == "Ease Out" then
		cam.CFrame = from:Lerp(to, 1-(1-t)^2)
	elseif style == "Ease In" then
		cam.CFrame = from:Lerp(to, t*t)
	elseif style == "Ease In-Out" then
		local v = t<0.5 and 2*t*t or 1-(-2*t+2)^2/2
		cam.CFrame = from:Lerp(to, v)
	elseif style == "Spring" then
		local spring = math.sin(t*math.pi)*0.3+t
		cam.CFrame = from:Lerp(to, math.clamp(spring,0,1))
	elseif style == "Velocity" then
		local angle = math.acos(math.clamp(from.LookVector:Dot(to.LookVector),-1,1))
		cam.CFrame = from:Lerp(to, math.clamp(t+angle*0.5,0.01,1))
	else
		cam.CFrame = from:Lerp(to, t)
	end
end)

	-- Aim/visual toggles
	cfg.camlock=false
	camlockTarget=nil
	cfg.silentAim=false
	cfg.esp=false
	toggleESP(false)

	-- Player/world toggles
	cfg.flightEnabled=false
	stopFlight()

	cfg.noclip=false
	disableNoclip()

	walkspeedEnabled=false
	stopWalkspeed()

	spinActive=false

	-- World visuals
	cfg.fogEnabled=false
	setFog(false)

	cfg.currentMaterial=nil
	resetMats()

	if resetTextureColors then
		resetTextureColors()
	end

	if cfg.selectedSkybox then
		applySky(nil)
	end
	cfg.selectedSkybox=nil

	cfg.rotatingSkybox=false
	if rotatingSkyConn then
		rotatingSkyConn:Disconnect()
		rotatingSkyConn=nil
	end

	-- Lighting
	cfg.fullbrightEnabled=false
	if fullbrightConn then
		fullbrightConn:Disconnect()
		fullbrightConn=nil
	end

	cfg.lightingMode=nil
	if lightingConn then
		lightingConn:Disconnect()
		lightingConn=nil
	end

	Lighting.Brightness=origL.Brightness
	Lighting.ClockTime=origL.ClockTime
	Lighting.GlobalShadows=origL.GlobalShadows

	-- Drawings
	if fovCircle then fovCircle.Visible=false end
	if camlockFOVCircle then camlockFOVCircle.Visible=false end

	-- Hide UI. RightCtrl brings it back.
	main.Visible=false
	blur.Size=0

	notify("Panic reset complete",2)
end

-- Patch render loop to handle camlock auto lock and KO check
RunService.Heartbeat:Connect(function()
	-- KO check: drop target if dead (only if KO check is enabled)
	if camlockTarget and cfg.camlockKOCheck then
		local hum=getHum(camlockTarget)
		if hum then
			-- Check for common KO indicators
			local char=camlockTarget.Character
			local bodyEffects=char and char:FindFirstChild("BodyEffects")
			
			if bodyEffects then
				-- Check multiple possible KO indicators
				local koValue=bodyEffects:FindFirstChild("K.O") or bodyEffects:FindFirstChild("KO") or bodyEffects:FindFirstChild("Knocked")
				
				-- Drop if health is 0 OR if K.O value exists and is true
				if hum.Health<=0 or (koValue and koValue.Value==true) then
					camlockTarget=nil
				end
			elseif hum.Health<=0 then
				camlockTarget=nil
			end
		else
			camlockTarget=nil
		end
	end
end)

local function formatTime(sec)
	local h = math.floor(sec / 3600)
	local m = math.floor((sec % 3600) / 60)
	local s = math.floor(sec % 60)

	return string.format("%02d:%02d:%02d", h, m, s)
end
